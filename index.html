In this blog, I will share Facebook-tagged Leetcode questions and my solutions to them. Though most likely not the exact problems that have been previously used, these questions should be a practical guide to understanding what you need to know for a Facebook interview.

https://leetcode.com/problems/minimum-remove-to-make-valid-parentheses/

class Solution:
  def minRemoveToMakeValid(self, s: str) -> str:
      pos = []
      stack = []
      ans = ""
        
      for i in range(len(s)):
          if s[i] == "(":
              stack.append(i)
          elif s[i] == ")":
              if len(stack) == 0:
                  pos.append(i)
              else:
                  stack.pop()
            
      while len(stack) > 0:
          pos.append(stack.pop())
            
      for i in range(len(s)):
          if i not in pos:
              ans += s[i]
                
      return ans

https://leetcode.com/problems/valid-palindrome-ii/

class Solution:
  def validPalindrome(self, s: str) -> bool:
        
      def helper(s, i, j):
          while i < j:
              if s[i] != s[j]:
                  return False
              i += 1
              j -= 1
          return True
          
      i = 0
      j = len(s) - 1
      while i < j:
          if s[i] != s[j]:
              return helper(s, i + 1, j) or helper(s, i, j - 1)
          i += 1
          j -= 1
      return True

https://leetcode.com/problems/lowest-common-ancestor-of-a-binary-tree-iii/

class Solution:
  def lowestCommonAncestor(self, p: 'Node', q: 'Node') -> 'Node':
      a = p
      b = q
      
      while a != b:
          if a == None:
              a = q
          else:
              a = a.parent
              
          if b == None:
              b = p 
          else:
              b = b.parent
